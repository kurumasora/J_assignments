#include <stdio.h>
 
int main (void) {
        int a[100][100] = {};
        int n;
        int m;
        int k;
 
        scanf("%d", &n);
        scanf("%d", &m);
 
        for (int i = 0; i < n; i++) {
                for (int j = 0; j < m; j++) {
                        scanf("%d", &a[i][j]);
                }
        }
 
        scanf("%d", &k);
 
 
        for (int j = 0; j < m; j++) {
                printf("%d ", a[k][j]);
        }
        printf("\n");
 
 
        return 0;
}
